	def _get_test_corpus(self):
		cats, docs = get_docs_categories_four()
		df = pd.DataFrame({'category': cats, 'text': docs})
		corpus = CorpusFromPandas(df, 'category', 'text', nlp=whitespace_nlp).build()
		return corpus
