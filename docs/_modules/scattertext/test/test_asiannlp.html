[docs]	def test_chinese(self):
		try:
			doc = chinese_nlp(self.chinese_text)
		except:
			return
		sent1 = doc.sents[0]
		self.assertEqual(str(sent1), u'总理 主持 召开 的 这场 座谈会 ， 旨在 听取 专家学者 和 企业界 人士 对 《 政府 工作 报告 （ 征求意见 稿 ） 》 的 意见建议 。')
		self.assertEqual(len(doc.sents), 11)
