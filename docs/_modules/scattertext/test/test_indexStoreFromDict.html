		         'it': 4,
		         'meyer': 5,

		         'name': 7,
		         'oscar': 8,

		idxstore = IndexStoreFromDict.build(vocab)
		for term, idx in vocab.items():
			self.assertEqual(idxstore.getidx(term), idx)
		self.assertEqual(idxstore.getnumvals(), len(vocab))
